f = open("GeoProcessingPython_list.txt","w+")
for a in alist:
    a1 = a.attrs['href']
    if (a1.find(".zip") > 0) or (a1.find(".pdf") > 0) or (a1.find(".py") > 0):
        print(url+a1)
        f.write(url+a1+"\n")
f.close()
