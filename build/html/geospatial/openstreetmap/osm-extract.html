def fast_iter(context, func):
    print('Process...')
    placement = 1
    for event, elem in context:
        placement += 1
        #if (placement>=10): break
        #print(placement,)

        func(elem)
        elem.clear()
        while elem.getprevious() is not None:
            del elem.getparent()[0]
    del context
